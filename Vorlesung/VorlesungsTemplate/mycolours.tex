  %\renewcommand{\mathcol}{}%
  \ifthenelse{%
    \boolean{colourif}}{%
    \color[#1]{#2,#3,#4}}{%
    \MULTIPLY{0.3}{#2}{\tmpr}
    \MULTIPLY{0.59}{#3}{\tmpg}
    \MULTIPLY{0.11}{#4}{\tmpb}
    \ADD{\tmpr}{\tmpg}{\tmprg}
    \ADD{\tmprg}{\tmpb}{\grayv}
    \color[gray]{\grayv}}}
